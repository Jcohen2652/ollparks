Cerveau logiciel

Le moteur de matching

8 fonctions atomiques + 1 score final + 1 action recommandée. Apprentissage continu sur l'historique des deals.

def calculate_final_score(besoin, bien, contact, history):
    parts = {
        "asset":          score_asset(besoin, bien),
        "geo":            score_geo(besoin, bien),
        "surface_budget": score_surface_budget(besoin, bien),
        "timing":         score_timing(besoin, bien),
        "signals":        score_signals(besoin),
        "history":        score_history(history),
        "contact":        score_contact(contact),
    }
    base = sum(parts.values())
    final = max(0, min(100, base - penalties(...)))
    return final, parts

def recommend_action(score):
    if score >= 80: return "appel_immediat"
    if score >= 60: return "envoi_fiche"
    if score >= 40: return "qualification"
    return "veille"
